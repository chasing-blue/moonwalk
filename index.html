git merge

git merge --no-commit --strategy recursive -X ours --no-ff 82384648ed

git checkout

git checkout -- src/something.txt /**< (discard the local change) */

git log

git log --name-status --oneline ./src/green.c
git log --invert-grep --grep="update" --no-merges --oneline my_branch/merge_1
git log --name-status --oneline --follow ./src/renamed.txt /**< for renamed file */

git config

git config --list --show-origin
git config --global alias.logad 'log --oneline --decorate --graph --all'
git config --system --list --show-origin

--local (Use this for stand-alone repositories, the configuration will be saved to the .git/ directory at the root of the repo itself)
--global (Use this for configuration settings you wish to apply to all repositories under a user account)
--system (Use this for configuration settings you wish to apply system-wide (To all users))

git grep

git grep --break --heading --show-function -n "function_name" -- *.[ch]

git reset

git reset --hard HEAD --recurse-submodules
